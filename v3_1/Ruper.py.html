 29 class Ruper:
 30     def __init__(self, vertices, segments, segmentsMark):
 31         self.vertices = []
 32         self.segments = {}
 33         self.segmentsMark = {}
 34         for vertex in vertices:
 35             self.AddVertex(vertex)
 36         for segment in segments:
 37             self.AddSegment(segment)
 38         for segmentMark in segmentsMark:
 39             self.AddSegmentMark(segmentMark[0], segmentMark[1])
 40         self.InitializeParameter()
 41 
 42     def InitializeParameter(self):
 43         self.stage = 0
 44         self.queueS = deque()
 45         self.queueT = deque()
 46         self.triangles = {}
 47         self.svRelation = {}
 48         self.vsRelation = {}
 49         self.checkedTriangles = []
 50 
 51     def InitializeDelaunay(self):
 52         delaunay = triPackage.triangulate({'vertices' : self.vertices}) #the usage of delaunay need pass the vertices
 53         for triangle in delaunay['triangles']:
 54             self.AddTriangle(triangle)
 55         return self.triangles
 56 
 57     def InitializeSegmentQueue(self):
 58         self.queueS.clear()
 59         for segment in self.segments:
 60             if self.IsEncroached(segment):
 61                 self.queueS.append(segment)
 62 
 63     def InitializeTriangleQueue(self):
 64         self.queueT.clear()
 65         for triangle in self.triangles:
 66             if self.IsSkinny(triangle):
 67                 self.queueT.append(triangle)
 68 
 69     def BruteForcePointLocation(self, vp):
 70         for triangle in self.triangles:
 71             va, vb, vc = self.vertices[triangle[0]], self.vertices[triangle[1]], self.vertices[triangle[2]]
 72             if InTriangle(vp, va, vb, vc):
 73                 return triangle
 74 
 75     def GetSegmentKey(self, segment):
 76         return tuple(sorted((segment[0], segment[1])))
 77 
 78     def GetTriangleKey(self, triangle):
 79         return tuple(sorted((triangle[0], triangle[1], triangle[2])))
 80 
 81     def InsertSV(self, segment, vertex):
 82         segmentKey = self.GetSegmentKey(segment)
 83         if self.svRelation.has_key(segmentKey) == False:
 84             self.svRelation[segmentKey] = {}
 85         self.svRelation[segmentKey][vertex] = True
 86 
 87     def InsertVS(self, vertex, segment):
 88         segmentKey = self.GetSegmentKey(segment)
 89         if self.vsRelation.has_key(vertex) == False:
 90             self.vsRelation[vertex] = {}
 91         self.vsRelation[vertex][segmentKey] = True
 92 
 93     def RemoveSV(self, segment, vertex):
 94         segmentKey = self.GetSegmentKey(segment)
 95         self.svRelation[segmentKey].pop(vertex)
 96 
 97     def RemoveVS(self, vertex, segment):
 98         segmentKey = self.GetSegmentKey(segment)
 99         self.vsRelation[vertex].pop(segmentKey)
100 
101     def AddVertex(self, vertex):
102         self.vertices.append(np.array(vertex))
103 
104     def AddSegment(self, segment):
105         self.segments[self.GetSegmentKey(segment)] = True
106 
107     def AddSegmentMark(self, segment, mark):
108         self.segmentsMark[self.GetSegmentKey(segment)] = mark
109 
110     def AddTriangle(self, triangle):
111         triangleKey = self.GetTriangleKey(triangle)
112         self.triangles[triangleKey] = True
113         self.checkedTriangles.append((triangle, 1))
114         for ind in range(3):
115             a, b, c = triangle[ind], triangle[(ind + 1) % 3], triangle[(ind + 2) % 3]
116             self.InsertSV((a, b), c)
117             self.InsertVS(a, (b, c))
118 
119     def DelVertex(self):
120         self.vertices.pop()
121 
122     def DelSegment(self, segment):
123         self.segments.pop(self.GetSegmentKey(segment))
124 
125     def DelSegmentMark(self, segment):
126         return self.segmentsMark.pop(self.GetSegmentKey(segment))
127 
128     def DelTriangle(self, triangle):
129         triangleKey = self.GetTriangleKey(triangle)
130         self.triangles.pop(triangleKey)
131         self.checkedTriangles.append((triangle, -1))
132         for ind in range(3):
133             a, b, c = triangle[ind], triangle[(ind + 1) % 3], triangle[(ind + 2) % 3]
134             self.RemoveSV((a, b), c)
135             self.RemoveVS(a, (b, c))
136 
137     def IsEncroached(self, segment):
138         segmentKey = self.GetSegmentKey(segment)
139         if self.segments.has_key(segmentKey) == False:
140             return False
141         if self.svRelation.has_key(segmentKey) == False or len(self.svRelation[segmentKey]) == 0:
142             return True
143         va, vb = self.vertices[segment[0]], self.vertices[segment[1]]
144         vm = (va + vb) * 0.5
145         r = GetDistance(va, vb) * 0.5
146         for ind in self.svRelation[segmentKey].keys():
147             v = self.vertices[ind]
148             if sgn(GetDistance(v, vm) - r) < 0:
149                 return True
150         return False
151 
152     def IsSkinny(self, triangle):
153         print triangle
154         triangleKey = self.GetTriangleKey(triangle)
155         if self.triangles.has_key(triangleKey) == False:
156             return False
157         va, vb, vc = self.vertices[triangle[0]], self.vertices[triangle[1]], self.vertices[triangle[2]]
158         vo = GetCircCenter(va, vb, vc)
159         if vo == None:
160             return True
161         r = GetDistance(va, vo)
162         d = np.min((GetDistance(va, vb), GetDistance(vb, vc), GetDistance(va, vc)))
163         if sgn(r / d - np.sqrt(2)) > 0:
164             return True
165         return False
166 
167     def RightSide(self, p, a, b):
168         segmentKey = self.GetSegmentKey((a, b))
169         if self.svRelation.has_key(segmentKey) == False or len(self.svRelation[segmentKey]) == 0:
170             return None
171         for v in self.svRelation[segmentKey].keys():
172             if v != p:
173                 return v
174 
175     def STest(self, p, a, b):
176         x = self.RightSide(p, a, b)
177         vp, va, vb = self.vertices[p], self.vertices[a], self.vertices[b]
178         if x == None:
179             if sgn(cross(vp, va, vb)) == 0:
180                 self.DelTriangle((p, a, b))
181             return
182         vx = self.vertices[x]
183         if InCircle(vp, va, vb, vx) == True:
184             self.os.AddFlipSequence(((p, a, b, x), 1))
185             self.DelTriangle((p, a, b))
186             self.DelTriangle((a, b, x))
187             self.AddTriangle((p, a, x))
188             self.AddTriangle((p, x, b))
189             self.STest(p, a, x)
190             self.STest(p, x, b)
191         else:
192             self.os.AddFlipSequence(((p, a, b, x), 0))
193 
194     def SwapTest(self, p, a, b, c):
195         self.STest(p, a, b)
196         self.STest(p, b, c)
197         self.STest(p, c, a)
198 
199     def RecoverTriangles(self):
200         checkedTriangles = self.checkedTriangles[:]
201         checkedTriangles.reverse()
202         for triangle in checkedTriangles:
203             if triangle[1] == -1:
204                 if self.triangles.has_key(self.GetTriangleKey(triangle[0])) == False:
205                     self.AddTriangle(triangle[0])
206             else:
207                 if self.triangles.has_key(self.GetTriangleKey(triangle[0])) == True:
208                     self.DelTriangle(triangle[0])
209         self.DelVertex()
210 
211 
212     def UpdateDelaunay(self):
213         self.checkedTriangles = []
214         p = len(self.vertices) - 1
215         triangle = self.BruteForcePointLocation(self.vertices[p])
216         a, b, c = triangle[0], triangle[1], triangle[2]
217         self.DelTriangle((a, b, c))
218         self.AddTriangle((p, a, b))
219         self.AddTriangle((p, b, c))
220         self.AddTriangle((p, c, a))
221         self.SwapTest(p, a, b, c)

224     def GetEncroachedS(self, triangleList):
225         encroachedS = []
226         for triangle in triangleList:
227             for ind in range(3):
228                 a, b = triangle[0][ind], triangle[0][(ind + 1) % 3]
229                 if self.IsEncroached((a, b)):
230                     encroachedS.append((a, b))
231         return encroachedS
232 
233     def GetSkinnyTriangles(self, triangleList):
234         skinnyT = []
235         for triangle in triangleList:
236             if self.IsSkinny(triangle[0]):
237                 skinnyT.append(triangle[0])
238         return skinnyT

240     def SplitSegment(self, segment):
241         va = self.vertices[segment[0]]
242         vb = self.vertices[segment[1]]
243         vm = (va + vb) * 0.5
244         count = len(self.vertices)
245 
246         self.AddVertex(vm)
247 
248         self.os.AddVertex(vm)
249 
250         self.DelSegment(segment)
251         self.AddSegment((segment[0], count))
252         self.AddSegment((segment[1], count))
253 
254         self.os.SplitSegment(segment)
255 
256         segmentMark = self.DelSegmentMark(segment)
257         self.AddSegmentMark((segment[0], count), segmentMark)
258         self.AddSegmentMark((segment[1], count), segmentMark)
259         self.UpdateDelaunay()
260 
261         encroachedS = self.GetEncroachedS(self.checkedTriangles)
262 
263         if self.IsEncroached((segment[0], count)) == True:
264             encroachedS.append((segment[0], count))
265         if self.IsEncroached((segment[1], count)) == True:
266             encroachedS.append((segment[1], count))
267 
268         for segment in encroachedS:
269             self.queueS.append(segment)
270 
271         self.os.AddEncroachedSegments(encroachedS)
272 
273         if self.stage == 3:
274             skinnyT = self.GetSkinnyTriangles()
275             for triangle in skinnyT:
276                 self.queueT.append(triangle)
277

279     def InsertCircleCenter(self, triangle):
280         va, vb, vc = self.vertices[triangle[0]], self.vertices[triangle[1]], self.vertices[triangle[2]]
281         vo = GetCircCenter(va, vb, vc)
282         if vo == None:
283             return
284         self.AddVertex(vo)
285         self.UpdateDelaunay()
286         encroachedS = self.GetEncroachedS(self.checkedTriangles)
287 
288         self.os.AddEncroachedSegments(encroachedS)
289 
290         if len(encroachedS) > 0:
291             self.os.operation = 2
292             self.RecoverTriangles()
293             for segment in encroachedS:
294                 self.queueS.append(segment)
295         else:
296             skinnyT = self.GetSkinnyTriangles(self.checkedTriangles)
297             for triangle in skinnyT:
298                 self.queueT.append(triangle)

301     def EliminateSegment(self):
302         while len(self.queueS) > 0:
303             segment = self.queueS.popleft()
304             if self.IsEncroached(segment):
305                 self.SplitSegment(segment)
306                 return True
307         return False
308 
309     def EliminateAngle(self):
310         while len(self.queueS) > 0:
311             segment = self.queueS.popleft()
312             if self.IsEncroached(segment):
313                 self.SplitSegment(segment)
314                 return True
315         while len(self.queueT) > 0:
316             triangle = self.queueT.popleft()
317             if self.IsSkinny(triangle):
318                 self.InsertCircleCenter(triangle)
319                 return True
320         return False

322     def CrossCount(self, u, v):
323         count = 0
324         for segment in self.segments:
325             va, vb = self.vertices[segment[0]], self.vertices[segment[1]]
326             state = GetSegIntersection(va, vb, u, v)
327             if state == None:
328                 return None
329             if state == True:
330                 count += self.segmentsMark[self.GetSegmentKey(segment)]
331         return count
332 
333     def RemoveOutside(self):
334         rmTriangles = []
335         for triangle in self.triangles:
336             va, vb, vc = self.vertices[triangle[0]], self.vertices[triangle[1]], self.vertices[triangle[2]]
337             u = (va + vb + vc) / 3.0
338             rm = False
339             for delta in (-1e-9, 0, 1e-9):
340                 v = np.array([1e5, u[1] + delta])
341                 count = self.CrossCount(u, v)
342                 if count != None and count % 2 == 0:
343                     rm = True
344             if rm == True:
345                 rmTriangles.append(self.GetTriangleKey(triangle))
346         for triangle in rmTriangles:
347             self.DelTriangle(triangle)

349     def NextStep(self):
350         print 'proccessing stage %d' % self.stage
351         if self.stage == 0:
352             self.os = OperationSequence(-1)
353             self.InitializeDelaunay()
354             self.InitializeSegmentQueue()
355             self.stage += 1
356         elif self.stage == 1:
357             self.os = OperationSequence(0)
358             if self.EliminateSegment() == False:
359                 self.stage += 1
360         elif self.stage == 2:
361             self.os = OperationSequence(3)
362             self.RemoveOutside()
363             self.stage += 1
364             self.InitializeTriangleQueue()
365         elif self.stage == 3:
366             self.os = OperationSequence(1)
367             if self.EliminateAngle() == False:
368                 self.stage += 1
369         return self.os
